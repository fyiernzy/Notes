INCLUDE Irvine32.inc

.data
prompt BYTE "Enter mark (0-100): ",0
result BYTE "Grade: ",0

.code
main PROC
     mov edx, OFFSET prompt 
     call WriteString
     call ReadInt
     call crlf
     
     .if eax >= 90
          mov al, 'A'
     .elseif eax >= 80
          mov al, 'B'
     .elseif eax >= 70
          mov al,'C'
     .elseif eax >= 60
          mov al,'D'
     .else
          mov al,'F'
     .endif
     
     mov edx, OFFSET result
     call WriteString
     call WriteChar
     call crlf
     
     exit
main ENDP
END main
