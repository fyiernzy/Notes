INCLUDE Irvine32.inc

.code
main PROC
     mov eax, 1
     mov edx, 1

L1:
     call WriteDec
     inc eax
     cmp eax, 9
     jne L1 ; if numbers not equal, go to L1

L2:
     call Crlf ; Create new line
     inc edx
     mov eax, edx
     cmp edx, 9
     jne L1 ; if numbers not equal, go to L1

     exit

main ENDP
END MAIN
