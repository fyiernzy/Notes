.data
     arr WORD 0,2,5,9,10 
     result WORD 0

.code
     main PROC
          mov ECX, LENGTHOF arr - 1 ; Iteration number the length of arr - 1
          mov ESI, OFFSET arr ; Set the pointer named ESI

          L1:
               mov AX, [ESI+2] ; Set the value of AX to the next value
               sub AX, [ESI] ; Subtract the AX by the previous value
               add result, AX ; Add the gap between the number to the result
               add ESI, 2 ; Increment the ESI by 2
          Loop L1   

          movzx ESI, result ; Set the value of ESI to result
          call DumpRegs
          
          exit
     main ENDP
END main
